#include <Arduino.h>
#include "WiFi.h"
#define LED 2


void setup() {
  Serial.begin(9600);

  WiFi.mode(WIFI_STA);
  WiFi.disconnect();
  delay(100);

  Serial.println("setup done");

}

void loop() {
  Serial.println("scn start");
  int n = WiFi.scanNetworks();
  Serial.println("scan done");
  if (n==0) {
    Serial.println("no networks found");
  } else {
    Serial.print(n);
    Serial.println(" networks found");
    for (int i = 0; i < n; i++){
      Serial.print(i + 1);
      Serial.print(": ");
      Serial.print(WiFi.SSID(i));
      Serial.print(" (");
      Serial.print(WiFi.RSSI(i));
      Serial.print(")");
      Serial.println((WiFi.encryptionType(i) == WIFI_AUTH_OPEN)?" ": "*");
      delay(10);
    }
  }
  Serial.println("");
  delay(5000);
}
